function letterSort(lang, letters) {
  letters.sort(new Intl.Collator(lang).compare);
  return letters;
}

console.log(letterSort('de', ['a','z','채']));
// expected output: Array ["a", "채", "z"]

console.log(letterSort('sv', ['a','z','채']));
// expected output: Array ["a", "z", "채"]
